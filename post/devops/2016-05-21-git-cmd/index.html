#!/bin/bash

COMMIT_MSG_FILE=$1
COMMIT_SOURCE=$2
SHA1=$3

RESULT="# Differences to be committed:"
while IFS= read -r line
do
  RESULT+="$IFS#    $line"
done <<< $(git diff --staged)

echo "$RESULT" >> $COMMIT_MSG_FILE
